    public void actualizar(){
      tblProductos.setModel(tableModel);
      while(tableModel.getRowCount()>0){
          tableModel.removeRow(0);
      }
      for(Ventas v: listaVentas){
          Object x[] = new Object[4];
          x[0]= v.getDescripcion();
          x[1]= v.getPrecio();
          x[2]= v.getCantidad();
          x[3]= v.getImporte();
          
          tableModel.addRow(x);
          
      }
      calularSubtotal();
      }
    public void calularSubtotal(){
        double subtotal = 0;
        for(Ventas venta:listaVentas){
            subtotal += venta.getImporte();
        }
        
        lblSubtotal.setText(String.valueOf(subtotal));
        
        double ivaPorcentaje = 0.16;
        double iva = subtotal * ivaPorcentaje;
        double total = subtotal + iva;
  
    
     
     lblSubtotal.setText(String.valueOf(subtotal));
     lblIva.setText(String.valueOf(iva));
     lblTotal.setText(String.valueOf(total));
    
            
}
    
    public void calculaPrecio(){
        precio = precios[cboProductos.getSelectedIndex()];
        cantidad= Integer.parseInt(spnCantidad.getValue().toString());
        lblPrecio.setText(String.valueOf(precio));
        lblImporte.setText(String.valueOf(precio*cantidad));
                
        
    }

    private void btnAgregarActionPerformed(java.awt.event.ActionEvent evt) {                                           
        // TODO add your handling code here:
        Ventas ventas = new Ventas();
        int indiceExistente = buscarProductoExistente(cboProductos.getSelectedIndex());

        if (indiceExistente != -1) {
        Ventas ventaExistente = listaVentas.get(indiceExistente);
        ventaExistente.setCantidad(ventaExistente.getCantidad() + cantidad);
        ventaExistente.setImporte(ventaExistente.getPrecio() * ventaExistente.getCantidad());
    } else {

        ventas.setId(cboProductos.getSelectedIndex());
        ventas.setDescripcion(cboProductos.getSelectedItem().toString());
        ventas.setPrecio(precio);
        ventas.setCantidad(cantidad);
        ventas.setImporte(precio*cantidad);
        
        listaVentas.add(ventas);
        }
        actualizar();
        
    }                                          
private int buscarProductoExistente(int idProducto) {
    for (int i = 0; i < listaVentas.size(); i++) {
        if (listaVentas.get(i).getId() == idProducto) {
            return i; // Retorna el índice si el producto ya existe
        }
    }
    return -1; // Retorna -1 si el producto no existe
}
